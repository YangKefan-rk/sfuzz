### 3. Relink a runnable emu with current SFUZ ABI support

```bash
cd ~/SFUZZ/sfuzz

CXX=clang++-18
VERILATOR_ROOT=/nfs/share/opt/verilator/share/verilator
LINKNAN=~/SFUZZ/LinkNan
REAL_MODEL_COMP="$LINKNAN/sim/emu/comp"
REAL_MODEL_GENERATED_SRC="$LINKNAN/build-cover-validate-r2/generated-src"
WORK=/tmp/sfuzz-litmus-probe2
mkdir -p "$WORK/relink" "$WORK/corpus"

COMMON_CXXFLAGS=(
  -std=c++17
  -DVERILATOR
  -DNUM_CORES=2
  -I"$VERILATOR_ROOT/include"
  -I"$VERILATOR_ROOT/include/vltstd"
  -I"$REAL_MODEL_COMP"
  -I"$LINKNAN/dependencies/difftest/config"
  -I"$REAL_MODEL_GENERATED_SRC"
  -I"$LINKNAN/dependencies/difftest/src/test/csrc/common"
  -I"$LINKNAN/dependencies/difftest/src/test/csrc/difftest"
  -I"$LINKNAN/dependencies/difftest/src/test/csrc/plugin/spikedasm"
  -I"$LINKNAN/dependencies/difftest/src/test/csrc/verilator"
  "-DNOOP_HOME=\"$LINKNAN\""
  -DREF_PROXY=NemuProxy
  -DEMU_THREAD=8
  -DFUZZER_LIB
  -DFUZZING
  -DLLVM_COVER
  -DFIRRTL_COVER
  -DVM_COVERAGE=1
  -fsanitize-coverage=trace-pc-guard
  -fsanitize-coverage=pc-table
)

"$CXX" "${COMMON_CXXFLAGS[@]}" -c "$LINKNAN/dependencies/difftest/src/test/csrc/common/main.cpp" -o "$WORK/relink/main.o"
"$CXX" "${COMMON_CXXFLAGS[@]}" -c "$LINKNAN/dependencies/difftest/src/test/csrc/common/ram.cpp" -o "$WORK/relink/ram.o"

"$CXX" -fuse-ld=lld-18 -fsanitize-coverage=trace-pc-guard -fsanitize-coverage=pc-table \
  "$WORK/relink/main.o" "$WORK/relink/ram.o" \
  "$REAL_MODEL_COMP/difftest-dpic.o" "$REAL_MODEL_COMP/SimJTAG.o" "$REAL_MODEL_COMP/common.o" "$REAL_MODEL_COMP/compress.o" "$REAL_MODEL_COMP/coverage.o" "$REAL_MODEL_COMP/device.o" "$REAL_MODEL_COMP/dut.o" "$REAL_MODEL_COMP/elfloader.o" "$REAL_MODEL_COMP/flash.o" "$REAL_MODEL_COMP/golden.o" "$REAL_MODEL_COMP/keyboard.o" "$REAL_MODEL_COMP/lightsss.o" "$REAL_MODEL_COMP/mpool.o" "$REAL_MODEL_COMP/perf.o" "$REAL_MODEL_COMP/query.o" "$REAL_MODEL_COMP/remote_bitbang.o" "$REAL_MODEL_COMP/sdcard.o" "$REAL_MODEL_COMP/uart.o" "$REAL_MODEL_COMP/vga.o" "$REAL_MODEL_COMP/difftest.o" "$REAL_MODEL_COMP/difftrace.o" "$REAL_MODEL_COMP/goldenmem.o" "$REAL_MODEL_COMP/refproxy.o" "$REAL_MODEL_COMP/spikedasm.o" "$REAL_MODEL_COMP/emu.o" "$REAL_MODEL_COMP/snapshot.o" "$REAL_MODEL_COMP/firrtl-cover.o" \
  "$REAL_MODEL_COMP/verilated.o" "$REAL_MODEL_COMP/verilated_dpi.o" "$REAL_MODEL_COMP/verilated_threads.o" "$REAL_MODEL_COMP/verilated_cov.o" "$REAL_MODEL_COMP/VSimTop__ALL.a" \
  -ldl -lrt -lpthread -lsqlite3 -lz -lzstd -latomic \
  target/release/libsfuzz.a \
  -o "$WORK/relink/emu"
```

### 4. Convert litmus ELF to SFUZ seed and run emu

Example for one litmus:

```bash
python3 ~/SFUZZ/sfuzz/scripts/make_sfuz_seed.py \
  --output /tmp/sfuzz-litmus-probe2/corpus/seed.sfuz \
  --core0-elf ~/SFUZZ/litmus-tests-riscv/generated-relax-model-linknan-elf/non-mixed-size/ATOMICS/RELAX/PodRWPX/LB+addr+popx-src/LB+addr+popx.elf

/tmp/sfuzz-litmus-probe2/relink/emu \
  --coverage FIRRTL.all \
  --fuzzing \
  --verbose \
  --max-iters 1 \
  --continue-on-errors \
  --corpus-input /tmp/sfuzz-litmus-probe2/corpus \
  -- --no-diff
```

Notes:

- `-- --no-diff` is required. Without it, the run stops earlier on missing `NEMU_HOME`.
- The current FIRRTL build uses `-DNUM_CORES=1`, while the generated litmus wrappers are 2-hart programs.
